    # 深拷贝

    1. 手动实现一个深拷贝
    /**
 * 思路：遍历对象属性，当前属性值为值类型或者null or undefined直接返回，否则再递归遍历属性值
 * 注：声明返回值的时候，要区分数组和对象
 */
 function deepClone(obj) {
     if(typeof obj != "object" || obj == null) {
         return obj
     }
     let result
     if(obj instanceof Array) {
         result = []
     } else {
         result = {}
     }
     for(let i in obj) {
         if(obj.hasOwnProperty(i)) {
             result[i] = deepClone(obj[i])
         }
     }
     return result
 }

    # Event Loop

    • 同步任务进入主线程， 异步任务进入Event Table（事件表）并注册函数
    • 当指定的事情完成时，Event Table会将这个函数移入Event Queue（事件队列）
    • 主线程内的任务执行完毕为空，回去Event Queue读取对应的函数，进入主线程执行
    • 上述过程会不断重复，也就是常说的Event Loop（事件循环）
    • JS引擎存在monitoring process进程，会持续不断的检查主线程执行栈是否为空，一旦为空，就回去Event Queue那里检查是否有等待被调用的函数

    # 手写建议jQuery考虑考虑拓展性

    class jQuery {
    constructor(selector) {
        this.selector = selector
        let elements = document.querySelectorAll(selector)
        for(let i = 0; i < elements.length; i++) {
            this[i] = elements[i]
        }
        this.length = elements.length
    }
    get(index) {
        return this[index]
    }
    each(callback) {
        for(let i = 0; i < this.length; i++) {
            callback(this[i])
        }
    }
    on(type, callback) {
        for(let i = 0; i < this.length; i++) {
            this[i].addEventListener(type, callback())
        }
    }
}

    # this

    • 作为普通函数被调用 谁调用this就是谁
    • 被apply call bind调用 传递的参数为this call,apply,bind三者的区别： call和apply基本用法一直，区别点在于传参，call可以铺开参数，apply只接收两个参数：this指向 和 数组
    • 作为对象的方法被调用 对象本身
    • 在class中被调用 实例本身
    • 箭头函数 this指向同函数外部的this this取什么值，是在函数执行的时候决定的，不是在定义的时候决定的

    # 原型中的this

    了解完了这些之后，我们解决一下上面提出的那些问题：

    1. this在不同场景中如何取值 同this描述部分内容
    2. 手写bind函数 bind函数与call和apply不同之处在于它不会立即执行方法，会返回一个方法，需要手动执行
    Function.prototype.myBind = function() {
    // 解析参数为数组
    const args = Array.prototype.slice.call(arguments)
    // 获取第一个参数即：this指向
    const first = args.shift()
    // 当前执行bind的函数
    const _this = this
    // 模仿bind返回一个方法
    return function() {
        // 执行原函数，返回执行结果
        return _this.apply(first, args)
    }
}
