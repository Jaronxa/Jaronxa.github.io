# 可选属性

interface Data {
    color: string;
    size?: number;      // 可选属性
}

function test(config: Data): {color: string, area: number} {
    let returnValue = {
        color: "",
        area: null
    }
    if(config.color) {
        returnValue.color = config.color
    }
    if(config.size) {
        returnValue.area = config.size * config.size
    }
    return returnValue;
}
test({
    color: "red",
    size: 10
})
/*
执行结果：
{
    color: 'red',
    size: 100
}
*/
test({
    color: "black"
})
/*
执行结果：
{
    color: 'black',
    area: null
}
*/

# 只读属性

接口标记 readonly 的属性值是不允许被修改的，定义处可以修改

interface Point {
    readonly x: number;
    readonly y: number;
}
let data: Point = {
    x: 10,
    y: 20
}
data.x = 5  // error

TS 有 ReadonlyArray<T> 类型，它与 Array<T> 相似，只是数组不允许被修改

let data1: number[] = [1,2,3,4,5,6];
let data2: ReadonlyArray<number> = data1;
data1[0] = 0
data2[0] = 0    // error

声明了 ReadonlyArray 的数组不可以直接赋值给其它变量，可以利用类型断言或者 any 类型

let arr: ReadonlyArray<number> = [1,2,3,4,5]
let data2: number[] = arr   // error
let data1: any = arr        // okay
let data3 = arr as number[] // okay

# 函数类型

interface searchFn {
    (color: string, size: number): boolean
}
let myFn: searchFn = function(color: string, size: number) {
    return size > 10
}
console.log(myFn(color: 'red', size: 9))    // false

# 可索引的类型

  • 可索引类型与接口描述函数类型差不多，我们可以描述那些通过索引得到的类型，比如 a[0] 或 o['name']
  • 可索引类型有一个索引签名，用于描述对象索引的类型，还有相应的返回值类型
interface StringArray {
    [index: number]: string
}
let myStrArr: StringArray = ["Jaron", "Jackson"]
console.log(myStrArr[0])    // Jaron

TS中支持两种索引签名：字符串和数字。可以同时使用，但数字索引返回值类型必须是字符串索引返回值类型的子类型。

class Animal {
    name: string
}
class Dog extends Animal {
    bread: string
}
interface Test {
    [x: string]: Animal;
    [y: number]: Dog;
}
